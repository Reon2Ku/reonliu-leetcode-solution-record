# 102.二叉树的层序遍历

## 题目描述

给你二叉树的根节点 `root` ，返回其节点值的 **层序遍历** 。 （即逐层地，从左到右访问所有节点）。

 

**示例 1：**

![img](https://assets.leetcode.com/uploads/2021/02/19/tree1.jpg)

```
输入：root = [3,9,20,null,null,15,7]
输出：[[3],[9,20],[15,7]]
```

**示例 2：**

```
输入：root = [1]
输出：[[1]]
```

**示例 3：**

```
输入：root = []
输出：[]
```

## 题解

自己想的题解，0ms，高贵的O(n)，也是发达了。唯一听取代码随想录的建议就是使用队列

确切地说，深度优先遍历使用栈，广度优先遍历使用队列

~~~
/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode() : val(0), left(nullptr), right(nullptr) {}
 *     TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
 *     TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left), right(right) {}
 * };
 */
class Solution {
public:
    vector<vector<int>> levelOrder(TreeNode* root) {
        vector<vector<int>> result;
        queue<TreeNode*> que;
        if(root==NULL) return result;
        que.push(root);
        que.push(NULL);
        TreeNode* node = que.front();
        while(!que.empty()&&que.front()!=que.back())
        {
            vector<int> temp;
            while(que.front()!=NULL)
            {
                node = que.front();
                if(node->left) que.push(node->left);
                if(node->right) que.push(node->right);
                que.pop();
                temp.push_back(node->val);
            }
            que.push(NULL);
            result.push_back(temp);
            que.pop();
        }
        return result;
    }
};
~~~

看了标准思路之后自己写了一下最简洁的写法，逻辑上也非常容易理解。反正比幽默深度优先遍历好理解多了

~~~
/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode() : val(0), left(nullptr), right(nullptr) {}
 *     TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
 *     TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left), right(right) {}
 * };
 */
class Solution {
public:
    vector<vector<int>> levelOrder(TreeNode* root) {
        vector<vector<int>> result;
        queue<TreeNode*> que;
        if(root==NULL) return result;
        que.push(root);
        while(!que.empty())
        {
            vector<int> vec;
            int size = que.size();
            for(int i=0;i<size;i++)
            {
                TreeNode* node = que.front();
                vec.push_back(node->val);
                que.pop();
                if(node->left) que.push(node->left);
                if(node->right) que.push(node->right);
            }
            result.push_back(vec);
        }
        return result;
    }
};
~~~

但是疑似还有一个递归法，我们也来敲一下。我自己完全想不到，但看又看得懂。

第一条经验：递归先写退出条件。

我觉得再遇到递归，要是以前见过类似的可以参考以前见过的写，要是没见过类似的就先争取看懂而不是一上来就要求自己写得出来

递归这个东西本来就抽象。迭代完全是要求自己想清楚了才能写得出的。

递归感觉更像是需要优先对于程序运行逻辑的整体进行分析解读之后才能设计得出来的算法

~~~
/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode() : val(0), left(nullptr), right(nullptr) {}
 *     TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
 *     TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left), right(right) {}
 * };
 */
class Solution {
public:
    void order(TreeNode* curr,vector<vector<int>>& vec,int depth)
    {
        //首先写递归退出条件.写递归第一行先写递归条件
        if(curr == NULL) return;
        if(vec.size()==depth) vec.push_back(vector<int>());
        vec[depth].push_back(curr->val);
        order(curr->left,vec,depth+1);
        order(curr->right,vec,depth+1);
    }
    vector<vector<int>> levelOrder(TreeNode* root) {
        int depth = 0;
        vector<vector<int>> result;
        order(root,result,depth);
        return result;
    }
};
~~~
