# 234. Palindrome Linked List

## question

Given the `head` of a singly linked list, return `true` *if it is a* *palindrome* *or* `false` *otherwise*.

 

**Example 1:**

![img](https://assets.leetcode.com/uploads/2021/03/03/pal1linked-list.jpg)

```
Input: head = [1,2,2,1]
Output: true
```

**Example 2:**

![img](https://assets.leetcode.com/uploads/2021/03/03/pal2linked-list.jpg)

```
Input: head = [1,2]
Output: false
```

 

**Constraints:**

- The number of nodes in the list is in the range `[1, 105]`.
- `0 <= Node.val <= 9`

 

**Follow up:** Could you do it in `O(n)` time and `O(1)` space?



## solution

暴力解

~~~
/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode() : val(0), next(nullptr) {}
 *     ListNode(int x) : val(x), next(nullptr) {}
 *     ListNode(int x, ListNode *next) : val(x), next(next) {}
 * };
 */
class Solution {
public:
    bool isPalindrome(ListNode* head) {
        vector<int> record;
        ListNode* curr = head;
        while(curr)
        {
            record.push_back(curr->val);
            curr = curr->next;
        }
        for(int i=0;i<record.size();i++)
        {
            if(record[i]!=record[record.size()-1-i]) return false;
        }
        return true;
    }
};
~~~

看了下评论区的思路后自己手撕的O(1)空间算法，似乎时间空间复杂度上都快了方法一一大截，几乎双百。我还以为会性能很差呢，结果还行。简述一下思路就是，用快慢指针找到链表的中点，并且在寻找过程中逐渐翻转前半部分。最后由找到的中点向左向右遍历对比，不等false，全等true

~~~
/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode() : val(0), next(nullptr) {}
 *     ListNode(int x) : val(x), next(nullptr) {}
 *     ListNode(int x, ListNode *next) : val(x), next(next) {}
 * };
 */
class Solution {
public:
    bool isPalindrome(ListNode* head) {
        if(!head) return true;
        else if(!head->next) return true;
        ListNode* fast = head;
        ListNode* slow = head;
        ListNode* temp = slow->next;
        ListNode* front = nullptr;
        while(fast->next && fast->next->next)
        {
            fast = fast->next->next;
            temp = slow->next;
            slow->next = front;
            front = slow;
            slow = temp;
        }
        temp = slow->next;
        slow->next = front;
        if(!fast->next && fast!=slow)
        {
            //cout<<"odd"<<endl;
            slow = slow->next;
        }
        while(slow)
        {
            if(slow->val!=temp->val) return false;
            slow = slow->next;
            temp = temp->next;
        }
        return true;

    }
};
~~~
