# 226.翻转二叉树

## 题目描述

给你一棵二叉树的根节点 `root` ，翻转这棵二叉树，并返回其根节点。

 

**示例 1：**

![img](https://assets.leetcode.com/uploads/2021/03/14/invert1-tree.jpg)

```
输入：root = [4,2,7,1,3,6,9]
输出：[4,7,2,9,6,3,1]
```

**示例 2：**

![img](https://assets.leetcode.com/uploads/2021/03/14/invert2-tree.jpg)

```
输入：root = [2,1,3]
输出：[2,3,1]
```

**示例 3：**

```
输入：root = []
输出：[]
```

## 题解

自己敲得（广度优先迭代法）。你疑似有一点陷入广度优先的怪圈里了，虽然这么也可以解，但我们不放跳脱出广度优先思维。

~~~
/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode() : val(0), left(nullptr), right(nullptr) {}
 *     TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
 *     TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left), right(right) {}
 * };
 */
class Solution {
public:
    TreeNode* invertTree(TreeNode* root) {
        queue<TreeNode*> que;
        if(root==NULL) return root;
        que.push(root);
        while(!que.empty())
        {
            int size = que.size();
            for(int i=0;i<size;i++)
            {
                TreeNode* node = que.front();
                que.pop();

                if(node->left&&node->right)
                {
                    que.push(node->left);
                    que.push(node->right);
                    //TreeNode* temp = node->right;
                    //node->right = node->left;
                    //node->left = temp;
                    swap(node->left,node->right);
                }
                else if(node->left)
                {
                    que.push(node->left);
                    node->right = node->left;
                    node->left = NULL;
                }
                else if(node->right)
                {
                    que.push(node->right);
                    node->left = node->right;
                    node->right = NULL;
                }
            }
        }
        return root;
    }
};
~~~

但其实这一题只要把所有节点遍历一遍即可，所以深度优先遍历明显也可以做这道题。自己敲得：

前序遍历

~~~
/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode() : val(0), left(nullptr), right(nullptr) {}
 *     TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
 *     TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left), right(right) {}
 * };
 */
class Solution {
public:
    TreeNode* invertTree(TreeNode* root) {
        stack<TreeNode*> sta;
        if(root==NULL) return root;
        sta.push(root);
        while(!sta.empty())
        {
            TreeNode* node = sta.top();
            sta.pop();
            if(node->right) sta.push(node->right);
            if(node->left) sta.push(node->left);
            swap(node->left,node->right); 
        }
        return root;
    }
};
~~~

递归法：自己写的

~~~
/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode() : val(0), left(nullptr), right(nullptr) {}
 *     TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
 *     TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left), right(right) {}
 * };
 */
class Solution {
public:
    void myInvert(TreeNode* node)
    {
        if(node==NULL) return;
        myInvert(node->left);
        myInvert(node->right);
        swap(node->left,node->right);
    }
    TreeNode* invertTree(TreeNode* root) {
        myInvert(root);
        return root;
    }
};
~~~
