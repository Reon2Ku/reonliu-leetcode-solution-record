# 718.最长重复子数组

## 题目描述

给两个整数数组 `nums1` 和 `nums2` ，返回 *两个数组中 **公共的** 、长度最长的子数组的长度* 。

 

**示例 1：**

```
输入：nums1 = [1,2,3,2,1], nums2 = [3,2,1,4,7]
输出：3
解释：长度最长的公共子数组是 [3,2,1] 。
```

**示例 2：**

```
输入：nums1 = [0,0,0,0,0], nums2 = [0,0,0,0,0]
输出：5
```

 

**提示：**

- `1 <= nums1.length, nums2.length <= 1000`
- `0 <= nums1[i], nums2[i] <= 100`

class Solution {
public:
    int findLength(vector<int>& nums1, vector<int>& nums2) {
        vector<int> dp(nums1.size(),0);
        int result = 0;
        //cout<<nums1.size()<<endl;
        for(int i=0;i<nums1.size();i++)
        {
             for(int j=0;j<nums2.size();j++)
             {
                if(nums1[i]==nums2[j])
                {
                    if(dp[i]==0) dp[i] = 1;
                    //cout<<"test i j = "<<i<<"  "<<j<<endl;
                    if(i!=0 && j!=0 && nums1[i-1]==nums2[j-1])
                    {

                        bool flag = true;
                        int count = 1;
                        for(int k = 0;k<dp[i-1];k++)
                        {
                            if(j-1-k<0)
                            {
                                flag = false;
                                break;
                            } 
                            if(nums1[i-1-k]!=nums2[j-1-k])
                            {
                                //cout<<"test i j = "<<i<<"  "<<j<<endl;
                                flag = false;
                                break;
                            }
                            else count++;
                        }
                        
                        if(flag) 
                        {
                            //cout<<"update dpi"<<endl;
                            dp[i] = max(dp[i],dp[i-1]+1);
                        }
                        else dp[i] = max(dp[i],count);
                    }
                }
             }
             //cout<<dp[i]<<endl;
             if(dp[i]>result) result = dp[i];
        }
        return result;
    }
};
~~~

关键还是在于dp数组的含义：

~~~
dp[i][j]表示：以nums1[i]和nums2[j]为结尾的字符串的最长公串的长度

~~~
class Solution {
public:
    int findLength(vector<int>& nums1, vector<int>& nums2) {
        vector<vector<int>> dp(nums1.size()+1,vector<int>(nums2.size()+1,0));
        int result = 0;
        for(int i=1;i<nums1.size()+1;i++)
        {
            for(int j=1;j<nums2.size()+1;j++)
            {
                //cout<<"dpij:"<<i<<"  "<<j<<endl;
                if(nums1[i-1]==nums2[j-1]) 
                {
                    dp[i][j] = dp[i-1][j-1]+1;
                }
                //else dp[i][j] = 0;

                if(dp[i][j]>result) result = dp[i][j];
                //cout<<"result:"<<dp[i][j]<<endl;
            }
        }
        return result;
    }
};
