# 50.Pow(x,n)

## 题目描述

实现 [pow(*x*, *n*)](https://www.cplusplus.com/reference/valarray/pow/) ，即计算 `x` 的整数 `n` 次幂函数（即，`xn` ）。

 

**示例 1：**

```
输入：x = 2.00000, n = 10
输出：1024.00000
```

**示例 2：**

```
输入：x = 2.10000, n = 3
输出：9.26100
```

**示例 3：**

```
输入：x = 2.00000, n = -2
输出：0.25000
解释：2-2 = 1/22 = 1/4 = 0.25
```

 

**提示：**

- `-100.0 < x < 100.0`
- `-231 <= n <= 231-1`
- `n` 是一个整数
- 要么 `x` 不为零，要么 `n > 0` 。
- `-104 <= xn <= 104`



## 题解

~~~
class Solution {
public:
    double myPow(double x, int n) {
        if(x==0) return 0;
        if(n==0) return 1;
        double result = 1;
        double coefficient = x;
        int size = n;
        if(n<0 && n!=INT_MIN)
        {
            coefficient = 1/x;
            size *= -1;
        }
        else if(n==INT_MIN) 
        {
            coefficient = 1/x;
            size = INT_MAX;
        }
        //cout<<"size = "<<size<<endl;

        int count1 = 0;
        int count2 = 1;
        if(size<=1073741824) 
        {
            while(count2<=size) 
            {
                count1++;
                count2*=2;
            }
        }
        else count1 = 31;
        
        //cout<<"count1 = "<<count1<<endl;
        vector<double> vec(count1,coefficient);
        vector<int> num(count1,1);
        cout<<vec[0]<<endl;
        for(int i=1;i<count1;i++)
        {
            vec[i] = vec[i-1]*vec[i-1];
            num[i] = num[i-1]*2;
            //cout<<vec[i]<<endl;
        }

        count1-=1;
        while(size>0)
        {
            size-=num[count1];
            if(size>=0) result*=vec[count1];
            else 
            {
                size+=num[count1];
                count1--;
            }
        }
        if(n==INT_MIN) result*=coefficient;
        return result;
    }
};
~~~
