# 34.在排序数组中查找元素的第一个和最后一个位置

## 题目描述

给你一个按照非递减顺序排列的整数数组 `nums`，和一个目标值 `target`。请你找出给定目标值在数组中的开始位置和结束位置。

如果数组中不存在目标值 `target`，返回 `[-1, -1]`。

你必须设计并实现时间复杂度为 `O(log n)` 的算法解决此问题。

 

**示例 1：**

```
输入：nums = [5,7,7,8,8,10], target = 8
输出：[3,4]
```

**示例 2：**

```
输入：nums = [5,7,7,8,8,10], target = 6
输出：[-1,-1]
```

**示例 3：**

```
输入：nums = [], target = 0
输出：[-1,-1]
```

 

**提示：**

- `0 <= nums.length <= 105`
- `-109 <= nums[i] <= 109`
- `nums` 是一个非递减数组
- `-109 <= target <= 109`



## 题解

摆烂解法：

~~~
class Solution {
public:
    vector<int> searchRange(vector<int>& nums, int target) {
        vector<int> result = {-1,-1};
        for(int i=0;i<nums.size();i++)
        {
            if(nums[i]==target)
            {
                result[0] = i;
                break;
            }
        }
        for(int i=nums.size()-1;i>=0;i--)
        {
            if(nums[i]==target)
            {
                result[1] = i;
                break;
            }
        }
        return result;
    }
};
~~~

class Solution {
public:
    vector<int> searchRange(vector<int>& nums, int target) {
        vector<int> result = {-1,-1};
        if(nums.size()<1) return result;
        int left = 0;
        int right = nums.size()-1;
        if(target<nums[0] || target>nums[nums.size()-1]) return result;
        int mid = (right+left)/2;
        while(right>left)
        {
            mid = (right+left)/2;
            if(nums[mid]==target) break;
            if(mid==left || mid==right) break;
            else if(nums[mid]<target) left = mid;
            else if(nums[mid]>target) right = mid;
        }
        //cout<<mid<<endl;
        if(nums[right]==target && right-left==1) mid = right;
        if(nums[mid]!=target) return result;
        left = mid;
        right = mid;
        while(left>=0)
        {
            if(nums[left]==target) 
            {
                result[0] = left;
                left--;
            }
            else break;
        }
        while(right<nums.size())
        {
            if(nums[right]==target)
            {
                result[1] = right;
                right++;
            }
            else break;
        } 
        return result;
    }
};

两次二分查找，最坏依旧是O(logn)

~~~
class Solution {
public:
    vector<int> searchRange(vector<int>& nums, int target) {
        vector<int> result = {-1,-1};
        if(nums.size()<1) return result;
        if(nums.size()==1) 
        {
            if(nums[0]==target) return {0,0};
            else return result;
        }
        int left = 0;
        int right = nums.size()-1;
        while(right>left)
        {
            int mid = (right+left)/2;
            if(nums[mid]==target)
            {
                if(mid==0||nums[mid-1]!=target)
                {
                    result[0] = mid;
                    left = mid;
                    break;
                }
                else right = mid;
            }
            else if(mid==left)
            {
                if(nums[right]==target) 
                {
                    return {right,right};
                }
                else return result;
            }
            else if(nums[mid]>target) right = mid;
            else if(nums[mid]<target) left = mid;
        } 
        //cout<<result[0]<<endl;
        right = nums.size()-1;

        while(right>left)
        {
            int mid = (right+left)/2;
            if(nums[mid]==target)
            {
                if(nums[mid+1]!=target)
                {
                    result[1] = mid;
                    break;
                }
                else if(left==mid)
                {
                    result[1] = left+1;
                    break;
                } 
                else left = mid;
            }
            
            else if(nums[mid]>target) right = mid;
            else if(nums[mid]<target) left = mid;
        } 
        return result;
    }
};
